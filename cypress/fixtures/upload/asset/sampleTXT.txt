
import assetDataTypeAttachmentsPage from '../../../pages/assetDataTypeAttachmentsPage';
import 'cypress-wait-until';

const assettype = new assetDataTypeAttachmentsPage();


And('User Clicks on Configuration menu item', () => {
  cy.clickButton(assettype.configurationSidePanelItem())
})

And('User Clicks on Asset data types tab', () => {
  cy.clickButton(assettype.assetDataTypesTab())
})


Given('User clicks on {string} data type', (AseetDatatype) => {

  cy.clickButton(assettype.typeOfAssetdatatype(AseetDatatype))

})

And('Clicks on edit in meatball options for any one of the {string} listed in horizontal rows', (datatype) => {

  cy.clickButtonForce(assettype.editOption(datatype));

})

And('Clicks on Cancel on pop up', () => {

  cy.clickButton(assettype.cancelOnPopUp());

});

And('User upload file to Upload Asset', () => {
  cy.clickButton(assettype.assetUploadSidePanelItem());
  cy.clickButton(assettype.uploadAssetsButton())
  cy.fileupload(assettype.browseButton(), "upload/Template_Cabinets_v1.xls");
  cy.clickButton(assettype.Uploadbutton());
  cy.wait(2000)
  cy.validateText(assettype.getfileNameonHeader(),"Template_Cabinets_v1.xls")
  
})
